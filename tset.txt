#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int main() {
	int a, b, c;
	int x, y = 100000000;
	int num[9];
	
	scanf("%d", &a);
	scanf("%d", &b);
	scanf("%d", &c);

	x = a * b * c;
	
	while (x / y == 0) {
		x %= y;
		y /= 10;
	}
	for (int i = 0; i < 9; i++) {
		num[i] = x / y;
		x %= y;
		if (y == 1)
			break;
		y /= 10;
	}
	for (int i = 0; i < 10; i++) {
		int cnt = 0;
		for (int j = 0; j < 9; j++) {
			if (i == num[j]) {
				cnt++;
			}
		}
		printf("%d\n", cnt);
	}
	return 0;
}
